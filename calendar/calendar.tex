\newcommand\Kalender[2]{%

\includegraphics[height=16cm, keepaspectratio]{#2}
\vfill{}
    {\centering\Huge \textbf{\monthname[#1]}}
    \begin{tikzpicture}
        \draw [line cap=round,red, thick] (0,-.75em) -- (0.97*\paperwidth,-.75em);
    \end{tikzpicture}

\vfill{}
    \begin{tikzpicture}
        \calendar[dates=2017-#1-01 to 2017-#1-last, month list, %
            every day/.style={font=\Large,}, %
            day xshift=1.7em] if (Sunday) [red];
    \end{tikzpicture}
\vfill{}
\clearpage
}
